How to write it
<!-- Trigger -->
<button data-open="my-modal" class="air-btn ace">Open</button>

<!-- Modal -->
<div class="air-modal" id="my-modal">
  <div class="air-modal-box ace">
    <div class="air-modal-head">
      <div><h3 class="air-modal-title">Title</h3></div>
      <button data-close class="air-modal-x">✕</button>
    </div>
    <div class="air-modal-body">Content</div>
    <div class="air-modal-foot">
      <button data-close class="air-btn ghost sm">Cancel</button>
      <button class="air-btn ace sm">Confirm</button>
    </div>
  </div>
</div>
